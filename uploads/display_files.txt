    <script>
        function redirectToPage(url) {
            window.location.href = url;
        }

        var username = '{{ session["username"] }}';
        console.log('Username:', username);

        function handleButtonClick(message) {
            if (message == "Logout"){
                alert(message);
                window.location.href = '/logout';
            }
        }

        var prevQuery = search_query.text;

        function performSearch() {
            var searchInput = document.getElementById('search_query');
            var resultsDropdown = document.getElementById('results');
            var searchForm = document.getElementById('searchForm');

            // Clear previous results
            resultsDropdown.innerHTML = '';
            
            if (search_query.value.trim() === '') {
                // Refresh the page when the search input is empty
                location.reload();
                return;
            }


            // Do the search and populate results
            fetch('/search', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json',
                },
                body: JSON.stringify({ query: searchInput.value }),
            })
            .then(response => response.json())
            .then(data => {
                if (data.results.length > 0) {
                    data.results.forEach(result => {
                        var option = document.createElement('option');
                        option.value = result;
                        option.text = new URL(result).pathname.split('/').pop();
                        resultsDropdown.add(option);
                    });

                    // Show the results dropdown
                    resultsDropdown.style.display = 'block';
                }
                else {
                    // Display a message when there are no results
                    var option = document.createElement('option');
                    option.value = '';
                    option.text = 'No results found';
                    resultsDropdown.add(option);
                    resultsDropdown.style.display = 'block';
                }

            });
        }

        document.addEventListener('change', function (event) {
            var resultsDropdown = document.getElementById('results');
            var selectedValue = new URL(resultsDropdown.value).pathname.split('/').slice(2).join('/');
            // Redirect to the selected URL
            if (selectedValue) {
                window.location.href = '/view/' + selectedValue;
            }
        });

        document.addEventListener('click', function(event) {
            var resultsDropdown = document.getElementById('results');
            var searchInput = document.getElementById('search_query');

            if (event.target !== resultsDropdown && event.target !== searchInput) {
                resultsDropdown.style.display = 'none';
            }
        });

    </script>
