    <script>
        // Toggle the display of writeSection and uploadSection based on the selected option
        document.addEventListener('DOMContentLoaded', function () {
            const writeOption = document.getElementById('write_option');
            const uploadOption = document.getElementById('upload_option');
            const writeSection = document.getElementById('writeSection');
            const uploadSection = document.getElementById('uploadSection');
            const fileTitleInput = document.getElementById('file_title');
            const docTitleInput = document.getElementById('doc_title');
            const briefDescription = document.getElementById('brief_description');
            const content = document.getElementById('content');
            const briefFileDescription = document.getElementById('brief_file_description');
    
            // Initial setup
            fileTitleInput.removeAttribute('required');
            docTitleInput.setAttribute('required', 'required');
            briefFileDescription.removeAttribute('required');
    
            writeOption.addEventListener('change', function () {
                writeSection.classList.remove('disabled-section');
                uploadSection.classList.add('disabled-section');
                fileTitleInput.removeAttribute('required');
                docTitleInput.setAttribute('required', 'required');
                briefFileDescription.removeAttribute('required');
            });
    
            uploadOption.addEventListener('change', function () {
                writeSection.classList.add('disabled-section');
                uploadSection.classList.remove('disabled-section');
                fileTitleInput.setAttribute('required', 'required');
                docTitleInput.removeAttribute('required');
                briefDescription.removeAttribute('required');
                content.removeAttribute('required');
                briefFileDescription.setAttribute('required', 'required');
            });
        });

        function redirectToPage(url) {
            window.location.href = url;
        }

        function handleButtonClick(message) {
            if (message == "Logout"){
                alert(message);
                window.location.href = '/logout';
            }
        }
        
    </script>
